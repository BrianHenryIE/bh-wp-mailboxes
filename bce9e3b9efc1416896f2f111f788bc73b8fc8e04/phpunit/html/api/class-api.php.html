475     * Shared implementation for the three remote email actions.
476     *
477     * Fires filter `bh_wp_mailboxes_remote_email_action_{$action}` with the email and resolved mailbox
478     * settings. A returning value of true signals that the action was handled.
479     * Regardless, updates the relevant post meta and inserts a log comment.
480     *
481     * @param string   $action One of: mark_read, mark_unread, delete_on_server.
482     * @param BH_Email $email  The email to act on.
483     *
484     * @throws Exception When expected email account / connection / remote coordinates are not found.
485     */
486    protected function perform_remote_email_action( string $action, BH_Email $email ): void {
487
488        $email_account = $this->get_email_account_for_email( $email );
489
490        if ( is_null( $email_account ) ) {
491            throw new Exception( 'failed to get BH_Email_Account for email ' . esc_html( $email->post_type ) );
492        }
493
494        $connection = $this->get_connection_for_email_account( $email_account );
495        $post_id    = $email->get_post_id();
496
497        if ( is_null( $connection ) ) {
498            throw new Exception( 'No connection found for ' . esc_html( $email_account->display_name ) );
499        }
500
501        // Remote read/delete actions live on Supports_Fetching: a connection that cannot be queried cannot
502        // perform them.
503        if ( ! ( $connection instanceof Supports_Fetching ) ) {
504            throw new Exception( 'Connection does not support remote actions for ' . esc_html( $email_account->display_name ) );
505        }
506
507        $this->set_connection_credentials( $connection, $email_account );
508
509        if ( is_null( $email->get_remote_coordinates() ) ) {
510            throw new Exception( 'No remote coordinates found for ' . esc_html( $email_account->display_name ) );
511        }
512
513        switch ( $action ) {
514            case 'mark_read':
515                try {
516                    $connection->set_is_marked_read( $email->get_remote_coordinates(), true );
517                    $this->email_repository->update( $email, is_remote_read: true );
518                    // Reversible change → info.
519                    $this->insert_email_log_note( $post_id, 'Marked as read on server', 'info' );
520                } catch ( Throwable ) {
521                    $this->insert_email_log_note( $post_id, 'Failed to mark as read on server.', 'error' );
522                }
523                break;
524            case 'mark_unread':
525                try {
526                    $connection->set_is_marked_read( $email->get_remote_coordinates(), false );

528                    // Reversible change → info.
529                    $this->insert_email_log_note( $post_id, 'Marked as unread on server', 'info' );
530                } catch ( Throwable ) {
531                    $this->insert_email_log_note( $post_id, 'Failed to mark as unread on server.', 'error' );
532                }
533                break;
534            case 'delete_on_server':
535                try {
536                    $connection->do_delete_on_server( $email->get_remote_coordinates() );
537                    $this->email_repository->update( $email, is_remote_deleted: true );
538                    // Intentional irreversible change → notice.
539                    $this->insert_email_log_note( $post_id, 'Deleted on server', 'notice' );
540                } catch ( Throwable ) {
541                    $this->insert_email_log_note( $post_id, 'Failed to delete email on server.', 'error' );
542                }
543                break;
544        }
545    }
